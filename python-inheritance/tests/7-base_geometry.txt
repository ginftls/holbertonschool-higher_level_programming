# Test cases for BaseGeometry

# integer_validator should accept positive integers
>>> bg = BaseGeometry()
>>> bg.integer_validator("my_int", 12)
>>> bg.integer_validator("width", 89)

# integer_validator should raise TypeError for non-integer values
>>> try:
...     bg.integer_validator("name", "John")
... except Exception as e:
...     print(f"[{e.__class__.__name__}] {e}")
[TypeError] name must be an integer

# integer_validator should raise ValueError for zero and negative values
>>> try:
...     bg.integer_validator("age", 0)
... except Exception as e:
...     print(f"[{e.__class__.__name__}] {e}")
[ValueError] age must be greater than 0

>>> try:
...     bg.integer_validator("distance", -4)
... except Exception as e:
...     print(f"[{e.__class__.__name__}] {e}")
[ValueError] distance must be greater than 0
